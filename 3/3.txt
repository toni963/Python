class Vozilo:

    __naziv=" "

    __god=0

    _brzina=0

    

    def __init__(self,Naziv,Godina):

        self.__naziv=Naziv

        self._brzina=0

        if Godina<1950:

            self.__god=1950

        elif Godina>2020:

            self.__god=2020

        else:

            self.__god=Godina

    def getGod(self):

        return self.__god

    def getBrzina(self):

        return self._brzina

    def getNaziv(self):

        return self.__naziv

    def ubrzaj(self,_brzina):

        self._brzina+=_brzina

        return "Brzina je "+str(self._brzina)

    def uspori(self,_brzina):

        self._brzina-=_brzina

        if self._brzina<0:

            self._brzina=0

        return "Brzina je "+str(self._brzina)

    def __init__(self,Naziv,Godina):

        self.__motorUG=False

    def UGmotor(self):

        self._brzina=0

    def uspori(self,x):

        if self.__motorUG==True:

        else:

            return "Motor je ugasen"

class KreditnaKartica:

    korisnik=""

    brKartice=""

    banka=""

    dug=0

    limit=0

    def _init_(self,KORISNIK,BROJKARTICE,BANKA,LIMIT):

        self.korisnik=KORISNIK

        self.brKartice=BROJKARTICE

        self.banka=BANKA

        self.limit=LIMIT

    def uplata(self,iznos):

        if self.dug-iznos>=0:

            self.dug-=iznos

        else:

            print("Dug iznosi manje od uplacenog.")

    def isplata(self,iznos):

        if self.dug+iznos<=self.limit:

            self.dug+=iznos

        else:

            return False

    def getK(self):

        return self.korisnik

    def getBR(self):

        return self.brKartice

    def getB(self):

        return self.banka

    def getD(self):

        return self.dug

    def getL(self):

        return self.limit

class PKK(KreditnaKartica):

    def _init_(self,KORISNIK,BROJKARTICE,BANKA,LIMIT,ks):

        super()._init_(KORISNIK,BROJKARTICE,BANKA,LIMIT)

        self.__ks=ks

    def isplata(self,x):

        if super().isplata(x)==False:

            self.dug+=5
